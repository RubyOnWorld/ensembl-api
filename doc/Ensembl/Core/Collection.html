def self.get_species_id(species)
  species = species.downcase
  meta = Meta.find_by_sql("SELECT * FROM meta WHERE LOWER(meta_value) = '#{species}'")[0]
  if meta.nil?
    return nil
  else
    return meta.species_id
  end
end
