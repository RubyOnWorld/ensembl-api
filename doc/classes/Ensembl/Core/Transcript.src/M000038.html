      def coding_region_cdna_start
        answer = 0
        
        exons = self.exons.sort_by{|e| e.seq_region_start}

        if self.translation.start_exon.strand == -1
          exons.reverse!
        end
        
        exons.each do |exon|
          if exon == self.translation.start_exon
            answer += self.translation.seq_start
            return answer
          else
            answer += exon.length
          end
        end
        
      end
