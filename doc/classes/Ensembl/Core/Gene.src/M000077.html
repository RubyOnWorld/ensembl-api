      def self.find_all_by_name(name)
        answer = Array.new
        xrefs = Ensembl::Core::Xref.find_all_by_display_label(name)
        xrefs.each do |xref|
          answer.push(Ensembl::Core::Gene.find_by_display_xref_id(xref.xref_id))
        end
      
        return answer
      end
