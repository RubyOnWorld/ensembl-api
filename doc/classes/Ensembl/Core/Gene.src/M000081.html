      def all_xrefs
        answer = Array.new
        answer.push(self.xrefs)
        self.transcripts.each do |transcript|
          answer.push(transcript.xrefs)
          if ! transcript.translation.nil?
            answer.push(transcript.translation.xrefs)
          end
        end
        answer.flatten!
        return answer
      end
