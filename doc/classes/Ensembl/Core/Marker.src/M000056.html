      def self.find_all_by_name(name)
        marker_synonyms = Ensembl::Core::MarkerSynonym.find_all_by_name(name)
        answers = Array.new
        marker_synonyms.each do |ms|
          answers.push(Ensembl::Core::Marker.find_all_by_marker_id(ms.marker_id))
        end
        answers.flatten!
        return answers
      end
